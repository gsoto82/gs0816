22           @Autowired
23           private ToolRepository toolRepository;

26           void setUp() {
27               // Optional: Clear the repository before each test (if needed)
28               toolRepository.deleteAll();
29   
30               // Load some test data into the repository
31               toolRepository.save(new Tool("CHNS", "Chainsaw", "Stihl", 1.49, true, false, true));
32               toolRepository.save(new Tool("LADW", "Ladder", "Werner", 1.99, true, true, false));
33               toolRepository.save(new Tool("JAKD", "Jackhammer", "DeWalt", 2.99, true, false, false));
34               toolRepository.save(new Tool("JAKR", "Jackhammer", "Ridgid", 2.99, true, false, false));
35           }
36   
37           @Test
38           @DisplayName("Verify Tools are Loaded into In-Memory Database")
39           void shouldLoadToolsIntoInMemoryDatabase() {
40               // Verify 'CHNS' tool exists
41               Optional<Tool> chainsaw = toolRepository.findById("CHNS");
42               assertThat(chainsaw).isPresent()
43                       .hasValueSatisfying(tool -> {
44                           assertThat(tool.getToolType()).isEqualTo("Chainsaw");
45                           assertThat(tool.getBrand()).isEqualTo("Stihl");
46                           assertThat(tool.getDailyCharge()).isEqualTo(1.49);
47                           assertThat(tool.isWeekdayCharge()).isTrue();
48                           assertThat(tool.isWeekendCharge()).isFalse();
49                           assertThat(tool.isHolidayCharge()).isTrue();
50                       });
51   
52               // Verify 'LADW' tool exists
53               Optional<Tool> ladder = toolRepository.findById("LADW");
54               assertThat(ladder).isPresent();
55   
56               // Verify 'JAKD' tool exists
57               Optional<Tool> jackhammerDewalt = toolRepository.findById("JAKD");
58               assertThat(jackhammerDewalt).isPresent();
59   
60               // Verify 'JAKR' tool exists
61               Optional<Tool> jackhammerRidgid = toolRepository.findById("JAKR");
62               assertThat(jackhammerRidgid).isPresent();
63           }
64   
65           @Test
66           @DisplayName("Verify Specific Tool Data is Loaded Correctly")
67           void shouldLoadSpecificToolDataCorrectly() {
68               // Fetch the Chainsaw tool and verify its properties
69               Tool chainsaw = toolRepository.findById("CHNS").orElseThrow();
70               assertThat(chainsaw).isNotNull();
71               assertThat(chainsaw.getToolType()).isEqualTo("Chainsaw");
72               assertThat(chainsaw.getBrand()).isEqualTo("Stihl");
73               assertThat(chainsaw.getDailyCharge()).isEqualTo(1.49);
74               assertThat(chainsaw.isWeekdayCharge()).isTrue();
75               assertThat(chainsaw.isWeekendCharge()).isFalse();
76               assertThat(chainsaw.isHolidayCharge()).isTrue();
77           }
